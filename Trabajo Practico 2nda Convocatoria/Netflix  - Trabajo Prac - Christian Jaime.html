    data = []
    with open(csv_file_path, 'r') as csv_file:

#Funcin para generar los datos usados en el grfico Sunburst con D3

import csv
import json

def convert_row_to_string(row, column_order):
    values = []
    for column in column_order:
        value = row.get(column, "")
        values.append(str(value))
    return '_'.join(values)

def convert_csv_to_js(csv_file_path, js_file_path, column_order):
    data = []
    with open(csv_file_path, 'r') as csv_file:
        csv_reader = csv.DictReader(csv_file)
        for row in csv_reader:
            converted_row = convert_row_to_string(row, column_order)
            data.append(converted_row)

    with open(js_file_path, 'w') as js_file:
        js_file.write("data =" + json.dumps(data, indent=4))

    print("CSV to JS conversion complete.")

# Example usage
csv_file_path = 'Netflix Userbase.csv'
js_file_path = 'sunburst/Netflix_Userbase_sunburst.js'
column_order = ['Country','Gender','Subscription Type','Device',]

convert_csv_to_js(csv_file_path, js_file_path, column_order)
